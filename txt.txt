name: Deploy

on:
  push:
    branches:
      - master  # Trigger deployment only on master branch

jobs:
  deploy:
    name: Deploy
    runs-on: ubuntu-latest
    if: github.ref == 'refs/heads/master'  # Ensure it's only triggered for master branch
    steps:
      - name: Start Deployment
        run: |
          echo "üöÄ Starting deployment process..."

      - uses: actions/checkout@v3

      - name: Set up PHP
        uses: shivammathur/setup-php@v2
        with:
          php-version: '8.1' # Adjust PHP version if necessary
          tools: composer

    
      - name: Verify Connection Details
        run: |
          echo "‚úÖ Checking deployment configuration..."
          if [ -z "${{ secrets.SERVER_HOST }}" ]; then
            echo "‚ùå Error: SERVER_HOST is not set"
            exit 1
          fi
          if [ -z "${{ secrets.SERVER_USERNAME }}" ]; then
            echo "‚ùå Error: SERVER_USERNAME is not set"
            exit 1
          fi
          if [ -z "${{ secrets.SERVER_SSH_KEY }}" ]; then
            echo "‚ùå Error: SERVER_SSH_KEY is not set"
            exit 1
          fi
          if [ -z "${{ secrets.SERVER_PORT }}" ]; then
            echo "‚ùå Error: SERVER_PORT is not set"
            exit 1
          fi

      - name: Deploy to production
        uses: appleboy/ssh-action@v0.1.10
        with:
          host: ${{ secrets.SERVER_HOST }}
          username: ${{ secrets.SERVER_USERNAME }}
          key: ${{ secrets.SERVER_SSH_KEY }}
          port: ${{ secrets.SERVER_PORT }}
          script: |
            echo "üìÇ Navigating to project directory..."
            cd /home/sendtrul/spa.sendtruly.com
            echo "‚¨áÔ∏è Pulling latest changes..."
            git pull origin master
            echo "üì¶ Installing dependencies..."
            composer install --no-interaction --prefer-dist --optimize-autoloader
            echo "üõ† Clearing cache..."
            php artisan cache:clear
            php artisan config:cache
            php artisan route:cache

      - name: Deployment Status
        run: echo "‚úÖ Deployment completed successfully!"
